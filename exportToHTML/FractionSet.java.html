import java.io.OutputStream;
import java.io.PrintWriter;
import java.util.ArrayList;
import java.util.Collections;
import java.util.Comparator;
import java.util.List;

public class FractionSet extends ArrayList<Fraction> {

    /**
     * constructor
     */
    public FractionSet() {
    }

    /**
     * print Fractions
     * @param out
     */
    void print(OutputStream out) {
        PrintWriter pw = new PrintWriter(out);
        for (Fraction i : this) {
            pw.print(i.toString() + " ");
        }
        pw.println();
        pw.close();
    }


    /**
     * get Max Fraction
     * @return other
     */
    Fraction getMax() {
        Fraction other = Collections.max(this, Comparator.comparingDouble(Fraction::getMean));
        return other;
    }

    /**
     * get Min Fraction
     * @return other
     */
    Fraction getMin() {
        Fraction other = Collections.min(this, Comparator.comparingDouble(Fraction::getMean));
        return other;
    }

    /**
     * count Fraction more than Fraction1
     * @param other
     * @return count
     */
    Integer getMoreThan(Fraction other) {
        int count = 0;
        for (Fraction i : this) {
            if (i.compare(other) > 0) count++;
        }
        return count;
    }

    /**
     * count Fraction less than Fraction1
     * @param other
     * @return count
     */
    Integer getLessThan(Fraction other) {
        int count = 0;
        for (Fraction i : this) {
            if (i.compare(other) < 0) count++;
        }
        return count;
    }

    /**
     * get All fractions
     * @return listOfFRactions
     */
    List<Fraction> getAllFracs(){
        return new ArrayList<>(this);
    }

}
